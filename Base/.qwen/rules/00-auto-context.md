```python
def start_session():
    # ШАГ 1: Автоматически найти CURRENT_CONTEXT.md
    context_path = ".qwen/CURRENT_CONTEXT.md"
    
    # ШАГ 2: Прочитать (если существует)
    if file_exists(context_path):
        context = read_file(context_path)
        
        # ШАГ 3: Извлечь ключевую информацию
        last_task = extract(context, "ГЛАВНАЯ ЗАДАЧА")
        status = extract(context, "Статус")
        stopped_at = extract(context, "Где остановились")
        
        # ШАГ 4: Сообщить пользователю
        print(f"✅ Контекст восстановлен!")
        print(f"📊 Последняя задача: {last_task}")
        print(f"🎯 Статус: {status}")
        print(f"📍 Где остановились: {stopped_at}")
    else:
        # Первая сессия (нет контекста)
        print("👋 Привет! Это первая сессия.")
        print("📄 Откройте AI_START_HERE.md для начала работы.")
```
